### 1. 【最核心】Template模板输入完全错误，模型根本不知道跟踪目标
跟踪模型的核心逻辑是：**用第一帧GT框裁剪出的目标区域作为Template模板，让模型学习要跟踪的目标特征**。但你的代码里，直接把第一帧的**全图**resize成128x128当模板，完全没有裁剪目标！

### 2. 预测框的坐标转换完全错误，位置/尺寸全乱了

- 模型输出的`w, h`是**log空间的相对比例**，不是直接的归一化尺寸，不能直接乘原图宽高；
- 你没有做「搜索图→原图」的坐标反向映射，直接把resize后的坐标套到原图上，导致预测框完全错位。

### 3. 没有初始化跟踪器的状态，模型无法时序跟踪

- 双模态的拼接顺序、通道数必须和模型训练时完全一致，否则输入特征完全错乱。

## 三、可直接复制的修复方案
### 第一步：先修复最核心的「Template模板裁剪」逻辑
这是跟踪的基础，必须先让模型知道要跟踪的目标是什么。
```python
# 新增：根据GT框裁剪Template目标区域
def crop_template(img, bbox, template_size=128, context_amount=0.5):
    """
    按照SOT标准流程裁剪模板区域
    :param img: 原始图像 (H,W,3)
    :param bbox: GT框 [x, y, w, h]
    :param template_size: 模板输出尺寸
    :param context_amount: 上下文比例（目标周围的背景区域）
    :return: 裁剪并resize后的模板图像
    """
    x, y, w, h = bbox
    # 计算带上下文的裁剪区域
    cx, cy = x + w/2, y + h/2
    crop_sz = np.sqrt(w * h) * (1 + context_amount)
    # 计算裁剪边界
    x1 = int(np.round(cx - crop_sz / 2))
    y1 = int(np.round(cy - crop_sz / 2))
    x2 = int(np.round(cx + crop_sz / 2))
    y2 = int(np.round(cy + crop_sz / 2))
    # 边界处理
    h_img, w_img = img.shape[:2]
    x1_pad = max(0, -x1)
    y1_pad = max(0, -y1)
    x2_pad = max(0, x2 - w_img)
    y2_pad = max(0, y2 - h_img)
    # 裁剪+填充
    img_crop = img[max(0, y1):min(h_img, y2), max(0, x1):min(w_img, x2)]
    if x1_pad + y1_pad + x2_pad + y2_pad > 0:
        img_crop = cv2.copyMakeBorder(img_crop, y1_pad, y2_pad, x1_pad, x2_pad, cv2.BORDER_CONSTANT)
    # resize到模板尺寸
    img_crop = cv2.resize(img_crop, (template_size, template_size))
    return img_crop

# 修复后的Template初始化
first_frame = data_list[0]
gt_bbox_first = first_frame['gt_bbox']
```

### 第二步：修复输入处理和坐标转换逻辑
```python
# 修复后的输入生成函数
def make_model_input_correct(rgb_img, tir_img, size):
    """
    正确的双模态输入处理
    :param rgb_img: 裁剪/resize后的RGB图像 (H,W,3)
    :param tir_img: 裁剪/resize后的TIR图像 (H,W,3)
    :param size: 输出尺寸
    :return: 模型输入tensor [1,6,size,size]
    """
    mean_rgb = torch.tensor([0.485, 0.456, 0.406]).view(1, 3, 1, 1).cuda()
    std_rgb = torch.tensor([0.229, 0.224, 0.225]).view(1, 3, 1, 1).cuda()
    mean_tir = torch.tensor([0.5, 0.5, 0.5]).view(1, 3, 1, 1).cuda()
    std_tir = torch.tensor([0.5, 0.5, 0.5]).view(1, 3, 1, 1).cuda()

    # 处理RGB
    rgb = torch.from_numpy(rgb_img).float().permute(2, 0, 1).unsqueeze(0).cuda() / 255.0
    rgb = F.interpolate(rgb, size=(size, size), mode='bilinear', align_corners=False)
    rgb_norm = (rgb - mean_rgb) / std_rgb

    # 处理TIR（单通道转3通道，专属归一化）
    if tir_img.ndim == 2:
        tir_img = np.stack([tir_img, tir_img, tir_img], axis=2)
    tir = torch.from_numpy(tir_img).float().permute(2, 0, 1).unsqueeze(0).cuda() / 255.0
    tir = F.interpolate(tir, size=(size, size), mode='bilinear', align_corners=False)
    tir_norm = (tir - mean_tir) / std_tir

    # 拼接双模态输入（和训练时一致：RGB在前，TIR在后）
    input_tensor = torch.cat([rgb_norm, tir_norm], dim=1)
    return input_tensor

# 修复后的预测框坐标转换
def convert_pred_bbox_correct(pred_bbox, orig_h, orig_w, search_size=256):
    """
    正确的OSTrack/ViPT预测框坐标转换
    :param pred_bbox: 模型输出 [cx, cy, w, h] (cx,cy是搜索图归一化坐标，w,h是搜索图归一化尺寸)
    :param orig_h: 原图高度
    :param orig_w: 原图宽度
    :param search_size: 搜索图尺寸
    :return: 原图上的xywh坐标 [x1, y1, w, h]
    """
    cx_n, cy_n, w_n, h_n = pred_bbox[:4]
    # 先转换到搜索图的像素坐标
    cx_px = cx_n * search_size
    cy_px = cy_n * search_size
    w_px = w_n * search_size
    h_px = h_n * search_size
    # 映射回原图坐标
    scale_x = orig_w / search_size
    scale_y = orig_h / search_size
    cx_orig = cx_px * scale_x
    cy_orig = cy_px * scale_y
    w_orig = w_px * scale_x
    h_orig = h_px * scale_y
    # 转成xywh
    x1 = cx_orig - w_orig / 2
    y1 = cy_orig - h_orig / 2
    return [x1, y1, w_orig, h_orig]
```

### 第四步：修复Score Map可视化逻辑，避免背景噪声掩盖目标
```python
# 修复后的Score Map归一化
def normalize_score_map(score_map, gt_bbox=None, search_size=256):
    """
    用目标区域的最大值做归一化，避免背景噪声掩盖目标
    :param score_map: 模型输出的score map [H,W]
    :param gt_bbox: 可选，GT框，用于限定目标区域
    :param search_size: 搜索图尺寸
    :return: 归一化后的score map
    """
    sm = score_map.copy()
    if gt_bbox is not None:
        # 用GT区域的最大值做归一化
        x, y, w, h = gt_bbox
        sm_h, sm_w = sm.shape
        x1 = int(max(0, (x / search_size) * sm_w))
        y1 = int(max(0, (y / search_size) * sm_h))
        x2 = int(min(sm_w, ((x+w) / search_size) * sm_w))
        y2 = int(min(sm_h, ((y+h) / search_size) * sm_h))
        target_max = sm[y1:y2, x1:x2].max()
        sm = sm / target_max if target_max > 1e-8 else sm
    else:
        # 全局归一化，但是抑制背景
        sm = (sm - sm.min()) / (sm.max() - sm.min() + 1e-8)
        sm = np.power(sm, 0.5)  # 开方增强目标对比度，抑制背景
    return np.clip(sm, 0, 1)
```
